138  bool IsFinished() {

140  if ( ! _queue.TryDequeue(out item)) return true;
141 
142  var path = item.Value;
143  var coords = path.PathStep.Hex.Coords;

149  "Dequeue Path at {0} w/ cost={1,4} at {2}; estimate={3,4}:{4,4}.",
150  coords, path.TotalCost, path.HexsideExit, item.Key >> 16, item.Key & 0xFFFF);
151 
152  if (item.Key>>16 < _getBestSoFar()

154  < _getBestSoFar()

164  void ExpandHex(DirectedPath path, Hexside hexside) {
165  var here = path.PathStep.Hex;
166  var there = here.Neighbour(hexside);
167  if (there != null && ! _closed.Contains(there.Coords) ) {
168  var cost = _stepCost(here, hexside, there);
169  if( (cost > 0)
170  ) {
171  if( (path.TotalCost+cost < _getBestSoFar() || ! _open.ContainsKey(there.Coords))
172  ) {
173  var newPath = _addStep(path, there, hexside, cost);
174  var key = Estimate(there, newPath.TotalCost);
175 
176  TraceFlags.FindPathEnqueue.Trace(" Enqueue {0}: estimate={1,4}:{2,4}",
177  there.Coords, key>>16, key & 0xFFFF);
178 
179  DirectedPath oldPath;
180  if ( ! _open.TryGetValue(there.Coords, out oldPath)) {
181  _open.Add(there.Coords, newPath);
182  _queue.Enqueue(key, newPath);
183  } else if (newPath.TotalCost < oldPath.TotalCost) {
184  _open.Remove(there.Coords);
185  _open.Add(there.Coords, newPath);
186  _queue.Enqueue(key, newPath);
187  }
188 
189  _setBestSoFar(newPath, Partner.GetPartnerPath(there.Coords));
190  }

226  public static DirectedPath MergePaths(DirectedPath fwd, DirectedPath rev) {
227  if (rev!=null) {
228  while (rev.PathSoFar!=null) {
229  var hexside = rev.PathStep.HexsideEntry;
230  var cost = rev.TotalCost - (rev = rev.PathSoFar).TotalCost;
231  fwd = fwd.AddStep(rev.PathStep.Hex, hexside, cost);
232  }
233  }

241  BidirectionalPathfinder(IHex start, IHex goal,
242  LandmarkCollection landmarks, HashSet<HexCoords> closed, Func<int> getBestSoFar
243  ) {
244  _start = start;
245  _getBestSoFar = getBestSoFar;
246 
247  _vectorGoal = goal.Coords.Canon - start.Coords.Canon;
248  _open = new Dictionary<HexCoords, DirectedPath>();
249  _closed = closed;
250  _queue = new HotPriorityQueue<DirectedPath>(16);
251 
252  _landmark = landmarks
253  .OrderByDescending(l => l.HexDistance(goal.Coords)-l.HexDistance(start.Coords))
254  .FirstOrDefault();
255  _heuristic = c => _landmark.HexDistance(c) - _landmark.HexDistance(start.Coords);
256 
257  TraceFlags.FindPathDetail.Trace(true, "Find path from {0} to {1}; vectorGoal = {2}",
258  start.Coords, goal.Coords, _vectorGoal);
259 
260  var path = new DirectedPath(goal);
261  _open.Add(goal.Coords, path);
262  _queue.Enqueue (0, path);
263  }
