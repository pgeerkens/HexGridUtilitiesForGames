133  public void Enqueue(HexKeyValuePair<TKey,TValue> item) {
134  _items.Add(item);
135  var child = Count-1;
136  var parent = (child-1) / 2;
137 
138  while (child > 0 && _items[parent] > _items[child]) {
139  var heap = _items[parent]; _items[parent] = _items[child]; _items[child] = heap;
140  child = parent;
141  parent = (child-1) / 2;
142  }

181  void MinHeapifyDown(int current) {
182 
183  int leftChild;
184  while ( (leftChild = 2*current + 1) < _items.Count) {
185 
186  // identify smallest of parent and both children
187  var smallest = _items[leftChild] < _items[current] ? leftChild
188  : current;
189  var rightChild = leftChild + 1;
190  if (rightChild < _items.Count && _items[rightChild] < _items[smallest])
191  smallest = rightChild;
192 
193  // if nothing to swap, ... then the tree is a heap
194  if (current == smallest) break;
195 
196  // swap smallest value up
197  var temp = _items[current];
198  _items[current] = _items[smallest];
199  _items[smallest] = temp;
200 
201  // follow swapped value down and repeat until the tree is a heap
202  current = smallest;
203  }
204  }
205  }
